% Fabrique un petit registre de sets pour éviter l'imbrication de tests.
\makeatletter
\newcommand{\DeclareFontSet}[2]{%
  \expandafter\gdef\csname setupfonts@#1\endcsname{#2}%
}
\newcommand{\UseFontSet}[1]{%
  \@ifundefined{setupfonts@#1}{%
    \PackageError{fontset}{Unknown FontSet '#1'}%
      {Declare it with \string\DeclareFontSet\space first. Valid options: lm, lm-sans, bonum, libertinus, pagella, termes, erewhon, schola.}%
  }{%
    \csname setupfonts@#1\endcsname
  }%
}
\makeatother

%
% Latin Modern (lm)
%

    Extension      = .otf,
    UprightFont    = *-regular,

    BoldItalicFont = *-bolditalic,
    SmallCapsFont = {lmromancaps10-regular},
