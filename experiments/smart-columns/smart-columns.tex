\ExplSyntaxOn

\seq_new:N \l__columnize_items_seq
\seq_new:N \l__columnize_temp_seq
\tl_new:N  \l__columnize_temp_tl
\tl_new:N  \l__columnize_tempb_tl
\box_new:N \l__columnize_measure_box
\dim_new:N \l__columnize_default_label_dim
\dim_new:N \l__columnize_labelsep_dim
\dim_new:N \l__columnize_maxwidth_dim
\dim_new:N \l__columnize_maxlabel_dim
\dim_new:N \l__columnize_current_label_dim
\dim_new:N \l__columnize_current_text_dim
\dim_new:N \l__columnize_temp_dim
\int_new:N \l__columnize_maxcols_int
\int_new:N \l__columnize_bestcols_int

\NewDocumentEnvironment { columnize } { O{3} +b }
  { \columnize_process:nn { #1 } { #2 } }
  { }

\cs_new_protected:Npn \columnize_reset:
  {
    \seq_clear:N \l__columnize_items_seq
    \dim_zero:N  \l__columnize_maxwidth_dim
    \dim_zero:N  \l__columnize_maxlabel_dim
    \dim_set:Nn  \l__columnize_labelsep_dim { .5em }
    \hbox_set:Nn \l__columnize_measure_box { \labelitemi }
    \dim_set:Nn  \l__columnize_default_label_dim { \box_wd:N \l__columnize_measure_box }
  }

\cs_new_protected:Npn \columnize_process:nn #1 #2
  {
    \begingroup
      \columnize_reset:
      \int_set:Nn \l__columnize_maxcols_int { #1 }
      \columnize_collect_items:n { #2 }
      \columnize_choose_columns:
      \columnize_typeset:
    \endgroup
  }

\cs_new_protected:Npn \columnize_collect_items:n #1
  {
    \seq_set_split:Nnn \l__columnize_temp_seq { \item } { #1 }
    \seq_pop_left:NN \l__columnize_temp_seq \l__columnize_temp_tl
    \seq_map_inline:Nn \l__columnize_temp_seq
      {
        \tl_set:Nn \l__columnize_temp_tl { ##1 }
        \tl_trim_spaces:N \l__columnize_temp_tl
        \tl_if_blank:VTF \l__columnize_temp_tl
          { }
          { \columnize_store_item:n { \l__columnize_temp_tl } }
      }
  }

\cs_new_protected:Npn \columnize_store_item:n #1
  {
    \tl_set:Nn \l__columnize_temp_tl { #1 }
    \tl_trim_spaces:N \l__columnize_temp_tl
    \tl_if_blank:VTF \l__columnize_temp_tl
      { \columnize_store_item_with_label:nn { } { } }
      {
        \tl_set:Nx \l__columnize_tempb_tl { \tl_head:N \l__columnize_temp_tl }
        \tl_if_eq:VnTF \l__columnize_tempb_tl { [ }
          { \exp_args:NV \columnize_store_item_with_optional:w \l__columnize_temp_tl \q_stop }
          { \columnize_store_item_with_label:nn { } { \l__columnize_temp_tl } }
      }
  }

\cs_new_protected:Npn \columnize_store_item_with_optional:w [#1] #2 \q_stop
  {
    \tl_set:Nn \l__columnize_temp_tl { #1 }
    \tl_trim_spaces:N \l__columnize_temp_tl
    \tl_set:Nn \l__columnize_tempb_tl { #2 }
    \tl_trim_spaces:N \l__columnize_tempb_tl
    \columnize_store_item_with_label:nn { \l__columnize_temp_tl } { \l__columnize_tempb_tl }
  }

\cs_new_protected:Npn \columnize_store_item_with_label:nn #1 #2
  {
    \columnize_measure_label:n { #1 }
    \columnize_measure_text:n { #2 }
    \dim_compare:nNnTF { \l__columnize_current_label_dim } > { \l__columnize_maxlabel_dim }
      { \dim_set_eq:NN \l__columnize_maxlabel_dim \l__columnize_current_label_dim }
      { }
    \dim_set:Nn \l__columnize_temp_dim
      {
        \l__columnize_current_label_dim +
        \l__columnize_labelsep_dim +
        \l__columnize_current_text_dim
      }
    \dim_compare:nNnTF { \l__columnize_temp_dim } > { \l__columnize_maxwidth_dim }
      { \dim_set_eq:NN \l__columnize_maxwidth_dim \l__columnize_temp_dim }
      { }
    \seq_put_right:Nn \l__columnize_items_seq { \columnize_item:nn { #1 } { #2 } }
  }

\cs_new_protected:Npn \columnize_measure_label:n #1
  {
    \tl_if_blank:nTF { #1 }
      { \dim_set_eq:NN \l__columnize_current_label_dim \l__columnize_default_label_dim }
      {
        \hbox_set:Nn \l__columnize_measure_box { #1 }
        \dim_set:Nn \l__columnize_current_label_dim { \box_wd:N \l__columnize_measure_box }
      }
  }

\cs_new_protected:Npn \columnize_measure_text:n #1
  {
    \hbox_set:Nn \l__columnize_measure_box { \strut #1 }
    \dim_set:Nn \l__columnize_current_text_dim { \box_wd:N \l__columnize_measure_box }
  }

\cs_new_protected:Npn \columnize_choose_columns:
  {
    \int_set:Nn \l__columnize_bestcols_int { 1 }
    \int_step_inline:nn { \l__columnize_maxcols_int }
      {
        \dim_set:Nn \l__columnize_temp_dim
          { \dim_eval:n { ( \linewidth - \int_eval:n { ##1 - 1 } \columnsep ) / ##1 } }
        \dim_compare:nTF { \l__columnize_maxwidth_dim <= \l__columnize_temp_dim }
          { \int_set:Nn \l__columnize_bestcols_int { ##1 } }
          { }
      }
  }

\cs_new_protected:Npn \columnize_typeset:
  {
    \setlength{\labelsep}{\dim_use:N \l__columnize_labelsep_dim}
    \setlength{\labelwidth}{\dim_use:N \l__columnize_maxlabel_dim}
    \setlength{\leftmargini}{\dim_eval:n { \l__columnize_maxlabel_dim + \l__columnize_labelsep_dim }}
    \setlength{\itemsep}{0pt}
    \setlength{\topsep}{0pt}
    \setlength{\parsep}{0pt}
    \begin{multicols}{ \int_use:N \l__columnize_bestcols_int }
      \begin{itemize}
        \seq_use:Nn \l__columnize_items_seq { }
      \end{itemize}
    \end{multicols}
  }

\cs_new_protected:Npn \columnize_item:nn #1 #2
  {
    \tl_if_blank:nTF { #1 }
      { \item \columnize_text_box:n { #2 } }
      { \item[ #1 ] \columnize_text_box:n { #2 } }
  }

\cs_new_protected:Npn \columnize_text_box:n #1
  { \mbox{ \strut #1 }\ignorespaces }
\ExplSyntaxOff
