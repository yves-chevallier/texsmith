\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{calc,backgrounds}
\AddToHook{shipout/background}{%
  \begin{tikzpicture}[remember picture,overlay]
    % marges du cadre
    \def\margin{1cm}
    \def\e{1cm} % "rayon" du coin arrondi

    % coins du cadre
    \coordinate (A) at ($(current page.north west) + (0.1,-0.1)$);
    \coordinate (B) at ($(current page.north east) + (-0.1,-0.1)$);
    \coordinate (C) at ($(current page.south east) + (-0.1,0.1)$);
    \coordinate (D) at ($(current page.south west) + (0.1,0.1)$);

    % points pour le coin arrondi en haut à droite
    \coordinate (Bdown) at ($(B)-(0,\e)$);   % juste en dessous de B
    \coordinate (Bleft) at ($(B)-(\e,0)$);   % juste à gauche de B
    \coordinate (Corner) at ($(B)-(\e,\e)$); % centre de l'arc de cercle
    % Cadre principal avec coin arrondi :
    \draw (A) -- (Bleft) -- (Bdown) -- (C) -- (D) -- cycle;
    \draw (Corner)
     .. controls ($(Corner)+(0.3*\e,0.1*\e)$) and ($(Bdown)+(-0.3*\e,-0.1*\e)$) ..
     (Bdown);
    \draw (Bleft)
     .. controls ($(Bleft)+(0.1*\e,-0.3*\e)$) and ($(Corner)+(-0.1*\e,0.3*\e)$) ..
     (Corner);     
    %\draw (Bleft) -- arc (0:90:\e) -- (Corner) -- (Bdown); 
  \end{tikzpicture}%
}
